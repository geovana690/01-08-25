Interatividade com addEventListener

Este texto pode mudar de cor!

Parágrafo escondido/visível ao clicar no botão!

Códigos JavaScript dos botões:


// Mudar a cor do texto
document.getElementById("btnCorTexto").addEventListener("click", function () {
  document.getElementById("textoAlvo").style.color = "blue";
});

// Exibir alerta
document.getElementById("btnAlerta").addEventListener("click", function () {
  alert("Você clicou no botão de alerta!");
});

// Mostrar/ocultar parágrafo
document.getElementById("btnMostrarOcultar").addEventListener("click", function () {
  const p = document.getElementById("paragrafoEscondido");
  p.style.display = (p.style.display === "block") ? "none" : "block";
});

// Mudar a cor de fundo da página
document.getElementById("btnCorFundo").addEventListener("click", function () {
  document.body.style.backgroundColor = "#ffeaa7";
});

// Resetar tudo
document.getElementById("btnResetar").addEventListener("click", function () {
  document.getElementById("textoAlvo").style.color = "black";
  document.getElementById("paragrafoEscondido").style.display = "none";
  document.body.style.backgroundColor = "#f9f9f9";
});
